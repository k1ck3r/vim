# debian
apt install python3 python3-pip vim-nox
pip install powerline-status

# centos
yum install python3 python3-pip vim-enhanced bash-completion -y && pip install --upgrade pip && pip3 install powerline-status

# pathogen install
mkdir -p .vim/{autoload,bundle}
git clone https://github.com/tpope/vim-pathogen.git /tmp/pathogen && cp /tmp/pathogen/autoload/pathogen.vim .vim/autoload/
cp vim/hacks/systemctl /etc/bash_completion.d/

# vim
cat vim/vimrc >> ./.vimrc
cat vim/bashrc >> ./.bashrc

# nice reverse search and coloring some unix utils (eg. ps, tail, lsof etc etc)
cp -r vim/.fzf* .
chmod +x ~/.fzf/bin/fzf
